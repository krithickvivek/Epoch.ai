class CurriculumAgent:
    def select_next_topic(self, student):
        unlocked = self.get_unlocked(student)
        return min(unlocked, key=lambda t: t.mastery)

    def adapt_difficulty(self, mastery):
        return clip(round(mastery * 5), 1, 5)

    def should_quiz(self, student):
        return student.accuracy > 0.7
